110 template <typename T, typename = std::enable_if_t<std::is_arithmetic_v<T>>>
111 DataRetrievalResult RetrieveVectorData(void* input_vector, T* output_vector) {
112  // Check the vector first.
113  DataRetrievalResult check_vector_result =
114  CheckVectorForVectorDataRetrieval(input_vector);
115  // return if not OK.
116  if (check_vector_result != DataRetrievalResult::Ok) {
117  return check_vector_result;
118  }
119  // Check if this is mixed vector or symbol vector
120  if (IsMixedVector(input_vector) || kdb_wrapper::GetQTypeId(input_vector) ==
121  q_types::q_type_id<std::string>) {
122  return DataRetrievalResult::NotNumericalVector;
123  }
124 
125  // Now copy the data over.
126  if (q_types::TryCopyDataByQTypeIdAndType<T>(
127  kdb_wrapper::GetQTypeId(input_vector),
128  kdb_wrapper::GetNumberOfVectorElements(input_vector),
129  kdb_wrapper::GetVector(input_vector), output_vector)) {
130  return DataRetrievalResult::Ok;
131  } else {
132  return DataRetrievalResult::InvalidQTypeId;
133  }
134 }
