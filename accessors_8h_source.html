92  int input_q_type_id) {
93  return // Check if C type is mapped directly to this id.
94  IsCTypeMappedToQTypeId<T>(input_q_type_id) ||
95  // Check if the ID is an int

214 template <typename T, int NQTypeId>
215 bool CopyDataByQTypeIQAndType(
217  int input_q_type_id,
219  std::size_t number_of_elements,
222  void* input_data,
224  T* output_data) {
225  // Check if the two q type id matches
226  if (input_q_type_id != NQTypeId) {
227  return false;
228  }
229  // Correctly typed input data
230  CTypeForQTypeId<NQTypeId>* typed_input_data =
231  reinterpret_cast<CTypeForQTypeId<NQTypeId>*>(input_data);
232 
233  // Copy
234  std::copy_n(typed_input_data, number_of_elements, output_data);
235 
236  return true;
237 }

243 template <typename T, int... NQTypeIds>
244 DataRetrievalResult TryCopyDataByQTypeIdAndType(
246  int input_q_type_id,
248  std::size_t number_of_elements,

269 template <typename T, typename = std::enable_if_t<std::is_arithmetic_v<T>>>
270 DataRetrievalResult RetrieveVectorData(void* input_vector, T* output_vector) {
271  // Check the vector first.
272  DataRetrievalResult check_vector_result =
273  CheckVectorForVectorDataRetrieval(input_vector);
274  // return if not OK.
275  if (check_vector_result != DataRetrievalResult::Ok) {
276  return check_vector_result;
277  }
278  // Check if this is mixed vector or symbol vector
279  if (IsMixedVector(input_vector) ||
280  GetQTypeId(input_vector) == q_type_id<std::string>) {
281  return DataRetrievalResult::NotNumericalVector;
282  }
283 
284  // Now copy the data over.
285  return TryCopyDataByQTypeIdAndType<
286  T, q_boolean_type_id, q_byte_type_id, q_short_type_id, q_int_type_id,
287  q_long_type_id, q_real_type_id, q_float_type_id, q_char_type_id,
288  q_timestamp_type_id, q_month_type_id, q_date_type_id, q_datetime_type_id,
289  q_timespan_type_id, q_minute_type_id, q_second_type_id, q_time_type_id>(
290  GetQTypeId(input_vector), GetNumberOfVectorElements(input_vector),
291  GetVector(input_vector), output_vector);
292 }
