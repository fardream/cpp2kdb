184 template <typename T, typename = std::enable_if_t<std::is_arithmetic_v<T>>>
185 DataRetrievalResult RetrieveVectorData(void* input_vector, T* output_vector) {
186  // Check the vector first.
187  DataRetrievalResult check_vector_result =
188  CheckVectorForVectorDataRetrieval(input_vector);
189  // return if not OK.
190  if (check_vector_result != DataRetrievalResult::Ok) {
191  return check_vector_result;
192  }
193  // Check if this is mixed vector or symbol vector
194  if (IsMixedVector(input_vector) || kdb_wrapper::GetQTypeId(input_vector) ==
195  q_types::q_type_id<std::string>) {
196  return DataRetrievalResult::NotNumericalVector;
197  }
198 
199  // Now copy the data over.
200  if (q_types::TryCopyDataByQTypeIdAndType<T>(
201  kdb_wrapper::GetQTypeId(input_vector),
202  kdb_wrapper::GetNumberOfVectorElements(input_vector),
203  kdb_wrapper::GetVector(input_vector), output_vector)) {
204  return DataRetrievalResult::Ok;
205  } else {
206  return DataRetrievalResult::InvalidQTypeId;
207  }
208 }
