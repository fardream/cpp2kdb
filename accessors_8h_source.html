282 constexpr bool IsCTypeMappedToQTypeId(
284  int input_q_type_id) {
285  return
286  // Not Error
287  input_q_type_id != q_error_type_id &&
288  // mapped id is input_q_type_id

295 constexpr bool IsQTypeIdInt(
297  int input_q_type_id) {
298  // If the input is negative, make it positive.
299  int positive_q_type_id =
300  input_q_type_id > 0 ? input_q_type_id : -input_q_type_id;
301  // Iterate through all the possible types, including int.
302  return positive_q_type_id == q_time_type_id ||
303  positive_q_type_id == q_month_type_id ||

345 template <typename T>
346 constexpr bool IsSameType(
348  int input_q_type_id) {
349  return // Check if C type is mapped directly to this id.
350  IsCTypeMappedToQTypeId<T>(input_q_type_id) ||
351  // Check if the ID is an int
352  (std::is_same_v<int, T> && IsQTypeIdInt(input_q_type_id)) ||
353  // Check if the ID is a std::int64_t
354  (std::is_same_v<std::int64_t, T> && IsQTypeIdInt64(input_q_type_id)) ||
355  // Check if the ID is a double
356  (std::is_same_v<double, T> && IsQTypeIdDouble(input_q_type_id)) ||
357  // Check if the ID is a string
358  (std::is_same_v<std::string, T> && IsQTypeIdString(input_q_type_id));
359 }
