365  return

367  input_q_type_id != q_error_type_id &&
368  // mapped id is input_q_type_id

376 constexpr bool IsQTypeIdInt(
378  int input_q_type_id) {
379  // If the input is negative, make it positive.
380  int positive_q_type_id =
381  input_q_type_id > 0 ? input_q_type_id : -input_q_type_id;
382  // Iterate through all the possible types, including int.
383  return positive_q_type_id == q_time_type_id ||
384  positive_q_type_id == q_month_type_id ||
385  positive_q_type_id == q_date_type_id ||
386  positive_q_type_id == q_minute_type_id ||
387  positive_q_type_id == q_second_type_id ||
388  positive_q_type_id == q_type_id<int>;
389 }

426 template <typename T>
427 constexpr bool IsSameType(
429  int input_q_type_id) {
430  return // Check if C type is mapped directly to this id.
431  IsCTypeMappedToQTypeId<T>(input_q_type_id) ||
432  // Check if the ID is an int
433  (std::is_same_v<int, T> && IsQTypeIdInt(input_q_type_id)) ||
434  // Check if the ID is a std::int64_t
435  (std::is_same_v<std::int64_t, T> && IsQTypeIdInt64(input_q_type_id)) ||
436  // Check if the ID is a double
437  (std::is_same_v<double, T> && IsQTypeIdDouble(input_q_type_id)) ||
438  // Check if the ID is a string
439  (std::is_same_v<std::string, T> && IsQTypeIdString(input_q_type_id));
440 }
